def include_stock_code(code):
    if len(code) < 5 or len(code) > 8:
        return False

    if 'TEST' in code:
        return False

    for c in code:
        if c.isnumeric():
            return True

    return False


def filter_transactions(df):
    df = df.loc[(df['Quantity'] > 0) & (~df['Customer ID'].isna())]
    df = df[df.apply(lambda x: include_stock_code(str(x['StockCode'])), axis=1)]  

    return df


def aggregate(features_df, labels_df):
    features_df = filter_transactions(features_df)
    labels_df = filter_transactions(labels_df)
    
    cutoff_date = features_df['InvoiceDate'].max()
    features_df['item_amount'] = features_df['Price'] * features_df['Quantity']
    features = features_df\
        .groupby('Customer ID')\
        .apply(lambda x: pd.Series(
            {
                'total_amount': x['item_amount'].sum(), 
                'num_orders': x['Invoice'].nunique(),
                'avg_value': x['item_amount'].sum()/x['Invoice'].nunique(),
                'customer_retention': (cutoff_date - x['InvoiceDate'].min()).days,
                'avg_lead_time': (cutoff_date - x['InvoiceDate'].min()).days/x['Invoice'].nunique(),
                'country': x['Country'].max(),
                'description': ' '.join(x['Description'].fillna('')),
            }))\
        .reset_index()

    labels = labels_df\
        .groupby('Customer ID')\
        .apply(lambda x: pd.Series(
            {
                'total_target_amount': (x['Price']*x['Quantity']).sum(), 
            }))\
        .reset_index()

    mergeddf = pd.merge(features, labels, on='Customer ID', how='left')
    mergeddf['total_target_amount'] = mergeddf['total_target_amount'].fillna(0)

    return mergeddf

We stack together 3 three-month period as the train set and set aside another three-month period as the test set. The subsequent twelve months for each set are used to calculate the label.

Modeling Pipeline

We set up a feature processing pipeline using scikit learn.

  • For numerical features, we scale them using RobustScaler.
  • For categorical features, we convert them using one-hot encoder.
  • For text features, we convert them to Glove embedding vectors.

Finally, we fit the data to a gradient boosting tree model for predictions.
